// Search function
searchBar.addEventListener('input', function() {
    const query = searchBar.value.toLowerCase();
    resultsContainer.innerHTML = '';

    if (query) {
        const matchedDiseases = findDiseases(query);

        if (matchedDiseases.length > 0) {
            matchedDiseases.forEach(disease => {
                const listItem = document.createElement('div');
                listItem.textContent = disease;
                listItem.className = 'search-result-item';
                listItem.style.padding = '10px';
                listItem.style.cursor = 'pointer';
                listItem.style.borderBottom = '1px solid #ddd';
                listItem.style.transition = 'background-color 0.2s ease-in-out';
                listItem.onmouseover = () => listItem.style.backgroundColor = '#f1f1f1';
                listItem.onmouseout = () => listItem.style.backgroundColor = 'transparent';
                listItem.onclick = () => {
                    searchBar.value = disease;
                    const petType = query.split(' ')[2];
                    window.location.href = `${petType}.html#${disease.replace(/\s+/g, '')}`;
                };
                resultsContainer.appendChild(listItem);
            });
        } 
    }
});

// Search button click event
searchButton.addEventListener('click', function() {
    const query = searchBar.value.toLowerCase();
    const matchedDiseases = findDiseases(query);

    if (matchedDiseases.length > 0) {
        const petType = query.split(' ')[2];
        window.location.href = `${petType}.html#${matchedDiseases[0].replace(/\s+/g, '')}`;
    } else {
        displayNoResult();
    }
});

// Helper function to find diseases
function findDiseases(query) {
    const match = query.match(/(\w+) in (\w+)/);
    if (match) {
        const symptom = match[1];
        const petType = match[2];

        if (diseaseData[petType]) {
            return Object.keys(diseaseData[petType]).filter(disease =>
                diseaseData[petType][disease].symptoms.some(sym => sym.toLowerCase().includes(symptom))
            );
        }
    }
    return [];
}

// Function to display 'Search not found'
function displayNoResult() {
    resultsContainer.innerHTML = '';
    const noResult = document.createElement('div');
    noResult.textContent = 'Search not found';
    noResult.style.padding = '10px';
    noResult.style.color = 'gray';
    noResult.style.textAlign = 'center';
    resultsContainer.appendChild(noResult);
}
